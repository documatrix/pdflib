else()

  if(cxx_std_20 IN_LIST CXX_FEATURES)
    set(CMAKE_CXX_STANDARD 20)
  endif()

  set (CMAKE_VALA_OPTS ${CMAKE_VALA_OPTS} -D LOG_DEBUG)
  set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -D LOG_DEBUG")

set(HAVE_CONFIG_H false)
foreach( CFLAGS ${DEPS_CFLAGS} )
  string(REGEX REPLACE "^-I" "" FILE_CHECK "${CFLAGS}")
  find_file(HAVE_FILE config.h PATHS ${FILE_CHECK} NO_DEFAULT_PATH)
  if(HAVE_FILE)
    set(HAVE_CONFIG_H true)
  endif()
endforeach( CFLAGS )

if(NOT HAVE_CONFIG_H)
  string(REGEX REPLACE "\n$" "" POPPLER_VERSION "${DEPS_poppler_VERSION}")
  string(REGEX REPLACE "0\\.(3[679]|4[3-9]|5[0-9]|60|61)\\.0" "0.\\1" POPPLER_VERSION "${POPPLER_VERSION}")
  execute_process(COMMAND wget -P ${CMAKE_SOURCE_DIR}/src https://raw.githubusercontent.com/documatrix/poppler/refs/tags/poppler-${POPPLER_VERSION}/glib/poppler-private.h)
  execute_process(COMMAND wget -P ${CMAKE_SOURCE_DIR}/src https://raw.githubusercontent.com/documatrix/poppler/refs/tags/poppler-${POPPLER_VERSION}/poppler/CairoOutputDev.h)
  execute_process(COMMAND wget -P ${CMAKE_SOURCE_DIR}/src https://raw.githubusercontent.com/documatrix/poppler/refs/tags/poppler-${POPPLER_VERSION}/poppler/SignatureInfo.h)
  execute_process(COMMAND wget -P ${CMAKE_SOURCE_DIR}/src https://raw.githubusercontent.com/documatrix/poppler/refs/tags/poppler-${POPPLER_VERSION}/poppler/HashAlgorithm.h)
  file(RENAME ${CMAKE_SOURCE_DIR}/src/config_template.h ${CMAKE_SOURCE_DIR}/src/config.h)
endif()

  ${CMAKE_VALA_OPTS} ${VAPIDIRS_ARGS}

  set(CMAKE_INSTALL_LIBDIR lib)
endif(NOT DEFINED CMAKE_INSTALL_LIBDIR)

  set(libdir       ${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_BINDIR})

  set(libdir       ${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_LIBDIR})

  ${CMAKE_CURRENT_BINARY_DIR}/${PDFLIB_NAME}.pc @ONLY)

  ${CMAKE_CURRENT_BINARY_DIR}/${DM_POPPLER_NAME}.pc @ONLY)

  # Install pdflib
